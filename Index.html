## Step 1: Import Required library import pandas as pd import numpy as np import matplotlib.pyplot as plt import seaborn as sns ## Step 2: Load Dataset from sklearn from sklearn.datasets import load_iris data = load_iris() df = pd.DataFrame(data.data, columns=data.feature_names) df['target'] = data.target df.head() ## Step 3: Check missing values df.isnull().sum() ## Step 4: Convert Into Independent Feature (X) and Dependent Feature(y) Or Target(y) X = df.drop('target', axis=1) y = df['target'] ## Step 5: Split Data into Train and Test from sklearn.model_selection import train_test_split X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42) ## Step 6: Train the Decision Tree Model from sklearn.tree import DecisionTreeClassifier model = DecisionTreeClassifier(random_state=42) model.fit(X_train, y_train) ## Step 7: Visualize the Tree from sklearn.tree import plot_tree plt.figure(figsize=(20,10)) plot_tree(model, feature_names=X.columns, class_names=str(np.unique(y)), filled=True) plt.show() ## Step 8:Make Predictions y_pred = model.predict(X_test) ## Step 9: Check Accuracy from sklearn.metrics import accuracy_score print("Accuracy Score:", accuracy_score(y_test, y_pred)) ## Step 10: Classification Report from sklearn.metrics import classification_report print("Classification Report:", classification_report(y_test, y_pred)) ## Step 11: Confusion Matrix shows that the model performs well on classifying from sklearn.metrics import confusion_matrix sns.heatmap(confusion_matrix(y_test, y_pred), annot=True, fmt='d', cmap='Blues') plt.title("Confusion Matrix") plt.show()
